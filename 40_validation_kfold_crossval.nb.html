Script to validate points in ReSurvey database using RS data (S2 + SatEmb + Landsat)

K-fold cross validation of best models

Alicia Valdés

11 marzo 2026

This R script is used to validate the points in the ReSurvey database using either RS indicators from S2 or Landsat (indices + phenology + canopy height), or bands from the Google Satellite Embedding dataset: https://developers.google.com/earth-engine/datasets/catalog/GOOGLE_SATELLITE_EMBEDDING_V1_ANNUAL#description Here, we perform a K-fold cross validation of the best models.

filter_data_Landsat <- function(data, variables) {
  data %>%
    mutate(EUNISa_1 = as.factor(EUNISa_1)) %>%
    # Remove rows with missing values in predictors
    dplyr::filter(if_all(all_of(variables), ~ !is.na(.))) %>%
    # Remove all rows with wrong values of indices (not between -1 and 1)
    dplyr::filter(NDVI_max <= 1 & NDVI_min >= -1) %>%
    # With Landsat data, many observations removed because of high/low EVI values
    # In Landsat_extraction, values of EVI outside of expected range were set as NA,
    # so better not to use EVI here
    # dplyr::filter(EVI_max <= 1 & EVI_min >= -1) %>%
    dplyr::filter(SAVI_max <= 1 & SAVI_min >= -1) %>%
    dplyr::filter(NDMI_max <= 1 & NDMI_min >= -1) %>%
    dplyr::filter(NDWI_max <= 1 & NDWI_min >= -1) %>%
    # Select only variables needed
    dplyr::select(PlotObservationID, EUNISa_1, all_of(variables), Lctnmth,
           valid_1) %>%
    # Keep only rows with differences > 0 in variables
    filter(if_all(any_of(variables[grepl("diff", variables)]), 
                  ~ .x > 0)) %>%
    # Select only GPS points
    dplyr::filter(Lctnmth == "Location with GPS" | 
                    Lctnmth == "Location with differential GPS")
}

# Compute a ROC object for each class (one-vs-all)
roc_list_SatEmb <- lapply(classes_SatEmb, function(cls) {
  # Binary labels: 1 for cls, 0 for others
  bin_obs <- as.integer(pred_df_SatEmb$obs == cls)
  # Probability for the class
  prob_cls <- pred_df_SatEmb[[cls]]
  # Compute ROC; set direction = "<" if needed depending on your scores
  pROC::roc(response = bin_obs, predictor = prob_cls, quiet = TRUE)
})
names(roc_list_SatEmb) <- classes_SatEmb
# Extract ROC curve data (TPR vs FPR) per class
roc_df_SatEmb <- lapply(names(roc_list_SatEmb), function(cls) {
  coords <- pROC::coords(roc_list_SatEmb[[cls]], ret = c("specificity", "sensitivity"), transpose = FALSE)
  data.frame(Class = cls,
             FPR = 1 - coords$specificity,
             TPR = coords$sensitivity)
}) %>% bind_rows()

roc_plot_SatEmb <- ggplot(roc_df_SatEmb, aes(x = FPR, y = TPR, color = Class)) +
  geom_line(linewidth = 0.75) +
  #scale_color_manual(values = cols) +
  geom_abline(slope = 1, intercept = 0, linetype = "dashed", color = "gray50") +
  labs(# title = "ROC curves (one-vs-all) by class",
       x = "False Positive Rate (1 - Specificity)",
       y = "True Positive Rate (Sensitivity)") +
  theme_minimal(base_size = 12)
